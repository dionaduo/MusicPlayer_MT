**一、歌词动态滚动实现** <sup class="quote">1</sup>
1. **歌词解析与存储**
   解析LRC格式歌词并存储为结构化数据：<sup class="quote">2</sup><sup class="quote">1</sup>
```typescript
// 歌词行结构
class LrcLine {
  lineStartTime: number = 0; // 行起始时间(ms)
  words: Array<{ text: string, startTime: number }> = []; // 逐词数据
}

// 解析增强型歌词（支持逐词）
processEnhancedLrc(lyric: string[]): void {
  const lrcList: LrcLine[] = [];
  const wordRegex = /<(\d{2}):(\d{2})\.(\d{2})>/g;

  lyric.forEach(line => {
    const lrcLine = new LrcLine();
    let match;

    while ((match = wordRegex.exec(line)) !== null) {
      const mm = parseInt(match, 10);
      const ss = parseInt(match, 10);
      const ms = parseInt(match, 10);
      const time = mm * 60000 + ss * 1000 + ms * 10;

      const textEndIndex = line.indexOf('<', match.index + 1);
      const text = textEndIndex === -1
        ? line.substring(match.index + match.length)
        : line.substring(match.index + match.length, textEndIndex);

      lrcLine.words.push({ text, startTime: time });
    }
    lrcList.push(lrcLine);
  });

  // 按时间排序
  lrcList.sort((a, b) => a.lineStartTime - b.lineStartTime);
  this.mLrcEntryList = lrcList;
}
```

2. **动画驱动歌词滚动**
   使用`animateTo`实现平滑滚动与文字渐变：
```typescript
@State currentLineIndex: number = 0;
@State currentWordIndex: number = 0;

// 歌词播放引擎
playLrc(duration: number) {
  animateTo({
    duration: duration,
    curve: Curve.Linear,
    onFinish: () => {
      // 切换到下一词或下一行
      if (this.currentWordIndex < this.mLrcEntryList[this.currentLineIndex].words.length - 1) {
        this.currentWordIndex++;
        const nextDuration = this.calculateWordDuration(this.currentLineIndex, this.currentWordIndex);
        this.playLrc(nextDuration);
      } else {
        this.currentLineIndex++;
        this.currentWordIndex = 0;
        if (this.currentLineIndex < this.mLrcEntryList.length) {
          const nextDuration = this.calculateWordDuration(this.currentLineIndex, 0);
          this.playLrc(nextDuration);
        }
      }
    }
  }, () => {
    // 动画执行期间更新透明度/位置
    this.textOpacity = 1;
  });
}
```

### 二、UI渲染优化

ForEach(this.mLrcEntryList, (line, lineIndex) => {
  Row() {
    ForEach(line.words, (word, wordIndex) => {
      Text(word.text)
        .fontSize(this.isActive(lineIndex, wordIndex) ? 24 : 18)
        .fontColor(this.isActive(lineIndex, wordIndex) ? '#FFFFFF' : '#AAAAAA')
        .opacity(this.isActive(lineIndex, wordIndex) ? 1 : 0.6)
    })
  }
  .margin({ bottom: 12 })
})

// 判断当前激活词
private isActive(lineIndex: number, wordIndex: number): boolean {
  return lineIndex === this.currentLineIndex &&
         wordIndex === this.currentWordIndex;
}
